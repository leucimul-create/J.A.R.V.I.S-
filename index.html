if (!textoLimpo) return; // Adiciona a pergunta do usuário na memória historicoConversa.push({ role: "user", content: textoLimpo }); exibirMensagemNoChat("user", textoLimpo); const indicadorId = exibirIndicadorDigitando(); try { const resposta = await fetch(URL_PROXY, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ mensagens: historicoConversa, }), }); if (!resposta.ok) { throw new Error(`Erro na API: ${resposta.status}`); } const dados = await resposta.json(); // CORREÇÃO AQUI: Captura o formato real vindo da Groq através do Worker const textoResposta = dados.choices[0].message.content; // Guarda a resposta do JARVIS na memória também historicoConversa.push({ role: "assistant", content: textoResposta }); removerIndicadorDigitando(indicadorId); exibirMensagemNoChat("assistant", textoResposta); return textoResposta; } catch (erro) { removerIndicadorDigitando(indicadorId); const mensagemErro = "Desculpe, senhor. Não consegui me conectar aos meus sistemas centrais."; exibirMensagemNoChat("assistant", mensagemErro); console.error("Erro ao falar com o JARVIS:", erro); return mensagemErro; } } // -------------------------------------------------------- // FUNÇÕES DE INTERFACE // -------------------------------------------------------- function exibirMensagemNoChat(autor, texto) { const container = document.getElementById("chat-container"); if (!container) return; const bolha = document.createElement("div"); // Mantém as classes msg-usuario e msg-jarvis compatíveis com o seu index.html bolha.className = autor === "user" ? "msg-usuario" : "msg-jarvis"; bolha.style.color = "white"; bolha.textContent = (autor === "user" ? "Você: " : "JARVIS: ") + texto; container.appendChild(bolha); container.scrollTop = container.scrollHeight; } function exibirIndicadorDigitando() { const container = document.getElementById("chat-container"); if (!container) return null; const indicador = document.createElement("div"); indicador.className = "msg-jarvis"; indicador.style.color = "#00d9ff"; indicador.textContent = "JARVIS está processando..."; indicador.id = "indicador-" + Date.now(); container.appendChild(indicador); container.scrollTop = container.scrollHeight; return indicador.id; } function removerIndicadorDigitando(id) { if (!id) return; const el = document.getElementById(id); if (el) el.remove(); }
